\begin{flalign*}
  \frac{\partial \mathbf{U}}{\partial t} +
  \frac{\partial \mathbf{E}}{\partial x} +
  \frac{\partial \mathbf{F}}{\partial y} +
  \frac{\partial \mathbf{G}}{\partial z} = 0
\end{flalign*}

\begin{flalign*}
  \frac{\partial \mathbf{U}_1}{\partial t} +
  \frac{\partial \mathbf{E}_1}{\partial \xi} +
  \frac{\partial \mathbf{F}_1}{\partial \eta} +
  \frac{\partial \mathbf{G}_1}{\partial \zeta} = 0
\end{flalign*}

\begin{flalign*}
  \mathbf{U}_1 & = \frac{\mathbf{U}}{J} \\
  \mathbf{E}_1 & = \frac{1}{J}\left(\mathbf{E}\xi_x + \mathbf{F}\xi_y + \mathbf{G}\xi_z\right) \\
  \mathbf{F}_1 & = \frac{1}{J}\left(\mathbf{E}\eta_x + \mathbf{F}\eta_y + \mathbf{G}\eta_z\right) \\
  \mathbf{G}_1 & = \frac{1}{J}\left(\mathbf{E}\zeta_x + \mathbf{F}\zeta_y + \mathbf{G}\zeta_z\right) \\
\end{flalign*}

\begin{flalign*}
J &= \frac{\partial(\xi,\eta,\zeta)}{\partial(x,y,z)} = \ninedet{\xi_x}{\xi_y}{\xi_z}{\eta_x}{\eta_y}{\eta_z}{\zeta_x}{\zeta_y}{\zeta_z} \\
&=
  \xi_x(\eta_y \zeta_z - \eta_z \zeta_y) -
  \xi_y(\eta_x \zeta_z - \eta_z \zeta_x) +
  \xi_z(\eta_x \zeta_y - \eta_y \zeta_x)
\end{flalign*}

\begin{flalign*}
\mathbf{E} &= \ilessv{E} \\
\mathbf{F} &= \ilessv{F} \\
\mathbf{G} &= \ilessv{G}
\end{flalign*}

\begin{flalign*}
  &\frac{\partial}{\partial t} \left(\frac{\mathbf{U}}{J}\right) +\\
  &\partialof{\xi} + \\
  &\partialof{\eta} + \\
  &\partialof{\zeta} = 0
\end{flalign*}


\begin{flalign*}
\mathbf{A} = \colvec{5}{\text{Continuity}}{\text{x-Momentum}}{\text{y-Momentum}}{\text{z-Momentum}}{\text{Energy}}
\end{flalign*}

\begin{flalign*}
  \mathbf{U} = \colvec{5}{\rho}{\rho u}{\rho v}{\rho w}{E_t}
\end{flalign*}

\begin{flalign*}
E_t = \rho \left(e+\frac{u^2+v^2+w^2}{2}\right)
\end{flalign*}

\begin{flalign*}
p = \rho R T
\end{flalign*}

\begin{flalign*}
(p + a \rho^2)\left(\frac{1}{\rho} - b\right) = R T
\end{flalign*}

\begin{flalign*}
  \mathbf{E_i} = \colvec{5}{\rho u}{\rho u^2+p}{\rho uv}{\rho uw}{(E_t+p)u}
\qquad
  \mathbf{F_i} = \colvec{5}{\rho v}{\rho uv}{\rho v^2+p}{\rho vw}{(E_t+p)v}
\qquad
  \mathbf{G_i} = \colvec{5}{\rho w}{\rho uw}{\rho vw}{\rho w^2+p}{(E_t+p)w}
\end{flalign*}

\begin{flalign*}
\mathbf{E_v} = \colvec{5}{0}{\tau_{xx}}{\tau_{xy}}{\tau_{xz}}{u\tau_{xx}+v\tau_{xy}+w\tau_{xz}-q_x}
\end{flalign*}

\begin{flalign*}
\mathbf{F_v} = \colvec{5}{0}{\tau_{xy}}{\tau_{yy}}{\tau_{yz}}{u\tau_{xy}+v\tau_{yy}+w\tau_{yz}-q_y}
\end{flalign*}

\begin{flalign*}
\mathbf{G_v} = \colvec{5}{0}{\tau_{xz}}{\tau_{yz}}{\tau_{zz}}{u\tau_{xz}+v\tau_{yz}+w\tau_{zz}-q_z}
\end{flalign*}

\begin{flalign*}
\tau_{xx} &= \frac{2}{3}\mu\left[2(\xi_xu_\xi + \eta_xu_\eta + \zeta_xu_\zeta) -
  (\xi_yv_\xi + \eta_yv_\eta + \zeta_yv_\zeta) -
  (\xi_zw)\xi + \eta_zw_\eta + \zeta_zw_\zeta)\right] \\
\tau_{yy} &= \frac{2}{3}\mu\left[2(\xi_yv_\xi + \eta_yv_\eta + \zeta_yv_\zeta) -
  (\xi_xu_\xi + \eta_xu_\eta + \zeta_zw_\zeta) -
  (\xi_zw)\xi + \eta_zw_\eta + \zeta_zw_\zeta)\right] \\
\tau_{zz} &= \frac{2}{3}\mu\left[2(\xi_zw_\xi + \eta_zw_\eta + \zeta_zw_\zeta) -
  (\xi_xu_\xi + \eta_xu_\eta + \zeta_zw_\zeta) -
  (\xi_yv_\xi + \eta_yv_\eta + \zeta_yv_\zeta)\right]
\end{flalign*}

\begin{flalign*}
\tau_{xy} &= \mu\left(\xi_yu_\xi + \eta_yu_\eta + \zeta_yu_\zeta + \xi_xv_\xi + \eta_xv_\eta + \zeta_xv_\zeta\right) \\
\tau_{xz} &= \mu\left(\xi_zu_\xi + \eta_zu_\eta + \zeta_zu_\zeta + \xi_xw_\xi + \eta_xw_\eta + \zeta_xw_\zeta\right) \\
\tau_{yz} &= \mu\left(\xi_zv_\xi + \eta_zv_\eta + \zeta_zv_\zeta + \xi_yw_\xi + \eta_yw_\eta + \zeta_yw_\zeta\right)
\end{flalign*}

\begin{flalign*}

\end{flalign*}
